    2. Create `server/routes/dxNewsSources/dxWorld.js` (CommonJS):
       ```javascript
       const Parser = require('rss-parser');
       const { extractCallsign, SOURCE_URLS } = require('../../utils/dxNewsMerge.js');

       const parser = new Parser({
         timeout: 10_000,
         headers: { 'User-Agent': 'OpenHamClock/3.13.1 (amateur radio dashboard)' },
       });

       function parseDxWorldFeed(feed) {
         if (!feed || !Array.isArray(feed.items)) return [];
         const items = [];
         for (const raw of feed.items) {
           const dt = new Date(raw.pubDate);
           if (isNaN(dt.getTime())) continue;  // skip items with unparseable dates
           items.push({
             id: `dxworld:${raw.guid || raw.link}`,
             title: raw.title || '',
             description: (raw.contentSnippet || raw.content || '').replace(/<[^>]+>/g, ' ').replace(/\s+/g, ' ').trim().slice(0, 200),
             url: raw.link,
             publishDate: dt.toISOString(),
             callsign: extractCallsign(raw.title),
             source: 'DX-WORLD',
             sourceUrl: SOURCE_URLS['DX-WORLD'],
           });
         }
         return items;
       }

       async function fetchDxWorld(ctx) {
         const feed = await parser.parseURL('https://dx-world.net/feed/');
         return { items: parseDxWorldFeed(feed) };
       }

       module.exports = { fetchDxWorld, parseDxWorldFeed };
       ```

    2. Create `server/routes/dxNewsSources/ng3k.js` (CommonJS):
       ```javascript
       const { SOURCE_URLS } = require('../../utils/dxNewsMerge.js');

       function reshapeDxpeditionCache(cacheData) {
         const list = cacheData?.dxpeditions;
         if (!Array.isArray(list)) return [];
         const items = [];
         for (const d of list) {
           if (!(d.isActive || d.isUpcoming)) continue;
           const desc = [d.dates, d.bands, d.modes].filter(Boolean).join(' · ');
           items.push({
             id: `ng3k:${d.callsign}`,
             title: `${d.callsign} — ${d.entity}`,
             description: desc,
             url: SOURCE_URLS.NG3K,
             publishDate: d.startDate,
             activityEndDate: d.endDate,
             callsign: d.callsign,
             source: 'NG3K',
             sourceUrl: SOURCE_URLS.NG3K,
           });
         }
         return items;
       }

       async function fetchNg3k(ctx) {
         const cacheData = ctx?.dxpeditionCache?.data;
         return { items: reshapeDxpeditionCache(cacheData) };
       }

       module.exports = { fetchNg3k, reshapeDxpeditionCache };
       ```
